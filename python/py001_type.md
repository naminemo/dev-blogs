# Type

```py
x = 10
y = 3.14
s = "Hello"
l = [1, 2, 3]
d = {'a': 1, 'b': 2}
b = True
n = None

print(type(x))  # <class 'int'>
print(type(y))  # <class 'float'>
print(type(s))  # <class 'str'>
print(type(l))  # <class 'list'>
print(type(d))  # <class 'dict'>
print(type(b))  # <class 'bool'>
print(type(n))  # <class 'NoneType'>
```
